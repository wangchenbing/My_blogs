  }

    params

    document.documentElement.scrollLeft || document.body.scrollLeft;

      return callback && callback();

      scrollLeft - averageStep <= xy.x ? xy.x : scrollLeft - averageStep;
    window.scrollTo(scrollLeft, scrollTop);

  };
